Fuehre diesen JavaScript-Code aus:
(function() {
  try {
    var chart = window.TradingViewApi._activeChartWidgetWV.value()._chartWidget;
    var sources = chart.model().model().dataSources();
    var results = {};
    sources.forEach(function(s) {
      if (!s.metaInfo) return;
      var name = '';
      try { name = s.metaInfo().description || s.metaInfo().shortDescription || ''; } catch(e) {}
      if (name.indexOf('Market Profile') === -1) return;
      var coll = s._graphics._primitivesCollection.dwglines.get('lines').get(false);
      var lines = [];
      coll._primitivesDataById.forEach(function(v) {
        if (v.y1 === v.y2) lines.push({y1: v.y1, x1: v.x1, x2: v.x2, ci: v.ci});
      });
      var max_x1 = Math.max.apply(null, lines.map(function(l){ return l.x1; }));
      var sess = lines.filter(function(l){ return l.x1 >= max_x1 - 3; });
      var poc_lines = sess.filter(function(l){ return l.ci === 0; });
      var va_lines  = sess.filter(function(l){ return l.ci === 1; });
      var poc = poc_lines.length ? Math.round(poc_lines[0].y1 * 100) / 100 : null;
      var vah = va_lines.length  ? Math.round(Math.max.apply(null, va_lines.map(function(l){ return l.y1; })) * 100) / 100 : null;
      var val = va_lines.length  ? Math.round(Math.min.apply(null, va_lines.map(function(l){ return l.y1; })) * 100) / 100 : null;
      var pos = (poc !== null && vah !== null && val !== null && vah !== val) ? Math.round((poc - val) / (vah - val) * 1000) / 1000 : null;
      var dist = sess.map(function(l) {
        return { price: Math.round(l.y1 * 100) / 100, count: l.x2 - l.x1, type: l.ci === 0 ? 'poc' : l.ci === 1 ? 'va' : 'tail' };
      }).filter(function(d){ return d.count > 0; }).sort(function(a,b){ return a.price - b.price; });
      results[name] = { poc: poc, vah: vah, val: val, mp_shape_pos: pos, tpo_distribution: dist };
    });
    return results;
  } catch(e) { return {error: e.message}; }
})()
